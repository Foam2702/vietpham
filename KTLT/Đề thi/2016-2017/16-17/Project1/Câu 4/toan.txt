#include  <iostream>
#include  <stdio.h>
using namespace std; 


int x[100];  int n, k;
void Xuatnghiem() {
    cout << "{ ";
    for (int i = 1; i <= k; i++)
        cout<< x[i]<<" ";
    cout<<"}" << endl;
}
void ToHop(int i) 
{
    for (int j = x[i - 1] + 1; j <= n - k + i; j++) 
    {
        x[i] = j;
        if (i == k)
            Xuatnghiem();
        else
            ToHop(i + 1);
    }
}
int main()
{
    cout << "n: ";
    cin >> n;
    cout << "{ }"<<endl;
    for(k=1;k<=n;k++)
        ToHop(1);
    system("pause");
}
